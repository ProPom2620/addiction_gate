    <div class="container">
        <h1>🛑 等一下！</h1>
        <p>你現在進去 YouTube 是為了特定的學習目標，還是單純想滑掉時間？</p>
        
        <input type="text" id="unlockInput" placeholder="請輸入「我要專注」以解除鎖定">
        
        <div class="buttons">
            <button id="goBtn" class="btn-go" onclick="grantAccess()">我已認清目標，進去吧</button>
            <button class="btn-leave" onclick="location.href='https://www.google.com'">算了，我不看了</button>
        </div>
    </div>

    <script>
        const input = document.getElementById('unlockInput');
        const btn = document.getElementById('goBtn');

        // 監聽輸入，只有打對字才會顯示按鈕
        input.addEventListener('input', (e) => {
            if (e.target.value === '我要專注') {
                btn.style.display = 'block';
            } else {
                btn.style.display = 'none';
            }
        });

        function grantAccess() {
            const urlParams = new URLSearchParams(window.location.search);
            // 拿到油猴傳過來的原網址，如果沒有就去 YT 首頁
            const returnUrl = urlParams.get('return') || 'https://www.youtube.com';
            const separator = returnUrl.includes('?') ? '&' : '?';
            // 帶著 pass=true 回去
            window.location.href = returnUrl + separator + 'pass=true';
        }
    </script>
